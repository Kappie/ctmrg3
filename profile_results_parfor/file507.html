        function destroyLocation(obj, entityLocation)
 292 
        %destroyLocation remove a location from storage
 293 
        %
 294 
        %  DESTROYLOCATION(OBJ, entityLocation)
 295 
        %
 296 
        % Removes all files associated with a task or a job. 
 297 
        %
 298 
        % ENTITYLOCATION   The 'location' of the task or job to be removed.
 299 
        
 300 
        % To destroy a location:
 301 
        %  * Delete files called entityLocation.ext where ext is any of
 302 
        %  the file extensions we've used to store data 
 303 
        %
 304 
        %  * If there is a folder called entityLocation, remove the folder. 
 305 
        %
 306 
        % The root storage location folder should never be removed. 
 307 

 308 
            % Don't bother if the storage is read only
 309 
            if obj.IsReadOnly
 310 
                error(message('parallel:cluster:FileStorageNotWritableChangeDir', obj.StorageLocation));
 311 
            end
 312 

 313 
            % Create a full location string
 314 
            fullLocation = [obj.StorageLocation filesep entityLocation];
 315 
            
 316 
            % Add .zip as a possible extension to delete because serializeForSubmission
 317 
            % might have created that file
 318 
            ext = [obj.Extensions ; '.zip'; '.lockstate'];
 319 

 320 
            % Turn off FileNotFound warning
 321 
            warningState1 = warning('off', 'MATLAB:DELETE:FileNotFound');
 322 
            warningState2 = warning('off', 'MATLAB:DELETE:Permission');
 323 
            % First try to remove the input files associated with the entity
 324 
            for i = 1:numel(ext)
 325 
                % It is possible that we will fail to delete a given file - don't
 326 
                % worry about this - we will delete the file when we delete the dir
 327 
                try
 328 
                    delete([fullLocation ext{i}]);
 329 
                catch e %#ok<NASGU>
 330 
                end
 331 
            end
 332 
            warning(warningState2);
 333 
            warning(warningState1);
 334 
            
 335 
            % Then remove any directories
 336 
            if exist(fullLocation, 'dir')
 337 
                % Again this might error - lack of permission or windows file locking
 338 
                try
 339 
                    % rmdir has different error behaviour depending on
 340 
                    % which output arguments are collected; we want to
 341 
                    % force all three to be generated. See g943440. 
 342 
                    [ok, msg, ~] = rmdir(fullLocation, 's');
 343 
                    if ~ok
 344 
                        dctSchedulerMessage( ...
 345 
                            5, ...
 346 
                            'Unable to delete job or task information folder. \n%s\n%s', ...
 347 
                            fullLocation, ...
 348 
                            msg);
 349 
                    end
 350 
                catch e %#ok<NASGU>
 351 
                    % Fail to delete - ignore this as we will send the directory to
 352 
                    % the deleteFile method of FileDeleter
 353 
                end
 354 
            end
 355 

 356 
            % If the directory still exists then lets ask the JVM to remove it when
 357 
            % it terminates
 358 
            if exist(fullLocation, 'dir') && usejava('jvm')
 359 
                import com.mathworks.toolbox.parallel.pctutil.io.FileDeleter;
 360 
                file = java.io.File(fullLocation);
 361 
                FileDeleter.deleteFile(file);
 362 
            end
 363 
            
 364 
            % Get the ID of the entity
 365 
            ID = iGetIDByName(entityLocation);
 366 
            
 367 
            % Finally remove references to this entity from the metadata file
 368 
            parent = iGetParentLocationFromEntityLocation(entityLocation);
 369 
            obj.removeEntityFromMetadata(parent, ID);
 370 
        end

        function [extensions, fileFormat] = getExtensionsForFields(~, type, names)
 433 

 434 
            import parallel.internal.cluster.FileStorage; 
 435 

 436 
            % MCOS property access is quite slow, and this function is
 437 
            % called frequently. We can reduce lookup times by using
 438 
            % persistent variables.
 439 
            persistent fieldToFormatIndexMaps
 440 
            persistent fileFormats
 441 
            
 442 
            if isempty(fieldToFormatIndexMaps)
 443 
                fieldToFormatIndexMaps = FileStorage.FieldToFormatIndexMaps;
 444 
                fileFormats = FileStorage.FileFormats;
 445 
            end
 446 

 447 
            % Attempt to get the map for this type of entity. 
 448 
            fieldToFormatIndexMap = fieldToFormatIndexMaps.(type);
 449 
            
 450 
            % If there's only one item in the cell array, we can go through
 451 
            % the fast scalar path, but we have to unpack the cell array
 452 
            % here, and repack it later. 
 453 
            DO_OPTIM = (numel(names) == 1) && iscell(names);
 454 
            if DO_OPTIM
 455 
                names = names{1};
 456 
            end
 457 
            
 458 
            % Treat the scalar case separately for performance reasons;
 459 
            % calling map.isKey, cell2mat and map.values are expensive, so
 460 
            % avoid all of those here.
 461 
            if DO_OPTIM || ischar(names)
 462 
                try
 463 
                    index = fieldToFormatIndexMap(names);
 464 
                catch %#ok<CTCH>
 465 
                    index = FileStorage.DefaultIndex;
 466 
                end
 467 

 468 
                format = fileFormats(index);
 469 
                extensions = format.Extension;
 470 
                fileFormat = format.FileFormat;
 471 
               
 472 
                if DO_OPTIM
 473 
                    % Repack the output into a single-element cell array. 
 474 
                    extensions = { extensions };
 475 
                    fileFormat = { fileFormat };
 476 
                end
 477 
                return
 478 
            end
 479 
            
 480 
            % Work out which field names we know about. 
 481 
            validNamesMask = fieldToFormatIndexMap.isKey(names);
 482 
            validNames = names(validNamesMask);
 483 

 484 
            % For each recognized name, find the corresponding extension
 485 
            % and format, and for the unrecognized ones, use a default
 486 
            % value. The indices variable will end up containing an index
 487 
            % into the ExtensionFormatPairs array for each field name that
 488 
            % was supplied.
 489 
            
 490 
            % maps.values() returns a cell array rather than a matrix, so we
 491 
            % have to use cell2mat to convert it to a valid index array.
 492 
            indices(validNamesMask) = cell2mat(fieldToFormatIndexMap.values(validNames));
 493 
            indices(~validNamesMask) = FileStorage.DefaultIndex;
 494 

 495 
            formats = fileFormats(indices); 
 496 
            extensions = {formats.Extension};
 497 
            fileFormat = {formats.FileFormat};
 498 
            
 499 
            % Ensure that output is the same shape as the input. 
 500 
            extensions = reshape(extensions, size(names));
 501 
            fileFormat = reshape(fileFormat, size(names));
 502 

 503 
        end
